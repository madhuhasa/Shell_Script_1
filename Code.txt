pipeline {
    agent any

    stages {
        stage('Checkout ') {
            steps {
                checkout([$class: 'GitSCM',
                branches: [[name: '*/master']], 
                doGenerateSubmoduleConfigurations: false,
                extensions: [[$class: 'RelativeTargetDirectory', 
                relativeTargetDir: 'shell_scripts']], 
                submoduleCfg: [], 
                userRemoteConfigs: [[url: 'https://github.com/madhuhasa/Shell_Script_1']]])

            }
            
        }
        stage('Executing') {
         steps {   
             
            sh "chmod  +x shell_scripts/first.sh"
            sh "chmod  +x shell_scripts/task2.sh"
          
          sh('shell_scripts/first.sh')   
          sh('shell_scripts/task2.sh')   
         }
        }
        stage('Archiving'){
         
         steps {
          archiveArtifacts "shell_scripts/*.sh"   
         }
         }
        stage('Final stage'){
            steps {
                echo "RESULT: ${currentBuild.currentResult}"
            } 
         } 
        stage('CentOS')
            agent {
                docker { image 'centos:3-alpine' }
            }
            steps {
                sh 'centos --version' }
          
        stage('Redhat')
            agent {
                docker { image 'redhat:3-alpine' }
            }
            steps {
                sh 'redhat --version'   }         
        }
        }
